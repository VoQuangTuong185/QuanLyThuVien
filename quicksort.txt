#include <iostream>
#include <algorithm>
using namespace std;

void partition(int arr[], int low, int high){
	int i = low;
	int j = high;
	int pivot = arr[(low+high)/2];
	int temp = 0;
	do{
	    while (arr[i] < pivot) i++;
	    while (arr[j] > pivot) j--;
	    if (i<=j){
	        temp = arr[i];
	        arr[i] =arr[j];
	        arr[j] = temp;
	        i++; j--;
	    }
	}
	while (i<=j);
    if(low < j) partition(arr, low, j);
	if(i < high) partition(arr, i, high);
}

void printArray(int arr[], int size){
	for (int i = 0; i < size; i++)
		cout << arr[i] << " ";
	cout << endl;
}

int main()
{
	int arr[] = {4, 2, 8, 3, 1, 5, 7,11,6};
	int size = sizeof(arr) / sizeof(int);
	cout<<"Before Sorting"<<endl;
	printArray(arr, size);
	partition(arr, 0, size - 1);
	cout<<"After Sorting"<<endl;
	printArray(arr, size);
	return 0;
}
